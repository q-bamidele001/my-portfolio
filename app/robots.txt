import { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: '*',
        allow: '/',
        disallow: '/admin/', // ← blocks Google from indexing your admin panel
      },
    ],
    sitemap: 'https://www.bamideleade.com.ng/sitemap.xml',
  };
}
